## Install

```bash
npm install --save reactjs-element-ojnd
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'reactjs-element-ojnd'
import 'reactjs-element-ojnd/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
